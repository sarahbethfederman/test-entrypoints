## Usage

```
import Tabs from '@lendi-ui/tabs';
```

```js

const StyledTabs= styled(Tabs)`
height: 50px;

`;

const FullStrechTab = styled(Tabs.Tab)`
  flex-basis: 50%;
  display: flex;  align-items: center;
  flex-direction: column;
`;

const Label = styled.span`
  margin-bottom: -4px;
  ${({ makeBold }) =>
    makeBold &&
    css`
      font-weight: bold;
    `}
`;

const StyledLock = styled(Lock)`  ${pr('xxs')} ${pt('xxs')}`;

class PlaygoundComponent extends React.Component{
  constructor() {
    this.state = {
      activeTabIndex: 0
    };
  }

  render() {
    return (
      <>
      <StyledTabs
          onChangeTabIndex={(activeTabIndex) => this.setState({ activeTabIndex })}
          activeTabIndex={this.state.activeTabIndex}
          isInverse
          activeBarStyles={{
              height: '5px',
            }}

        >
        <FullStrechTab icon={<StyledLock color="secondary.500" />}>
            <Label makeBold={0 === this.state.activeTabIndex}>Active Applications</Label>
          </FullStrechTab>
          <FullStrechTab icon={<StyledLock color="secondary.500" />}>
            <Label makeBold={1 === this.state.activeTabIndex}>InActive Applications</Label>
        </FullStrechTab>
        </StyledTabs >
      </>
    );
  }
 
}

render(<PlaygoundComponent/>)
```
