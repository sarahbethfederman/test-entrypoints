import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import Modal from '@lendi-ui/modal';
```

```jsx
<Modal show={false} size="lg" onHide={() => {}}>
  <Modal.Content>
    <span>
      You already started an application on<br />
      8 October 2018<br />
      Would you like to continue?
    </span>
  </Modal.Content>
  <Modal.Footer>
    <Button variant="primary" isFullWidth onClick={ ()=> {} }>
      Continue application
    </Button>
    <Button variant="secondary" isFullWidth onClick={ ()=>{} }>
      Start a new application
    </Button>
  </Modal.Footer>
</Modal>
```
