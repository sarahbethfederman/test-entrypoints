import { PropTable } from '@lendi-ui/doc-utils';

## Usage

```
import Overlay from '@lendi-ui/overlay';
```

```jsx
  <Overlay show={false} onHide={() => { alert('onHide action')}} aria-label="testlabel">
    <div>This is a demo</div>
  </Overlay>
```
