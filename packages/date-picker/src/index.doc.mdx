import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import DatePicker from '@lendi-ui/DatePicker';

// This is a controlled component, fields are limited to numerical
// characters only when working with proper changeHandlers and values.
// See the examples page for a working implementation.
```

```jsx
<DatePicker
  dayOnChange={() => {}}
  monthOnChange={() => {}}
  yearOnChange={() => {}}
/>
```
