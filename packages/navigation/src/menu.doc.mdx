import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import { Menu } from '@lendi-ui/navigation';
```

```jsx
<Menu menuTitle='Loan Options' isOpen={true} onClick={() => {}}>
  <MenuItem itemHeader='Workshop your deal' badge={<StatusBadge variant='success' hasIcon={false} size='sm' statusText="COMPLETED" />}  />
  <MenuItem itemHeader='Lender selection' badge={<StatusBadge variant='warn' hasIcon={false} size='sm' statusText="INCOMPLETED" />}  />
</Menu>        
```

## Properties

<PropTable>
  <PropTable.Entry name="menuTitle" required type="string" description="The header title of Menu component."/>
  <PropTable.Entry name="isOpen" type="boolean" defaultValue="false" description="The state of content div is whether open or close."/>
  <PropTable.Entry name="onClick" required type="(menuTitle:string) => void" description="The function that change the state of content div is whether open or close."/>
  <PropTable.Entry name="children" type="React.ReactNode" description="The content of Menu component."/>
  <PropTable.Entry name="className" defaultValue=""type="string" defaultValue="" description="The custom style for Menu component."/>
</PropTable>
