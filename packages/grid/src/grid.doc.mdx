# grid

A grid.

## Mixin

```
import { grid } from '@lendi-ui/grid';
```

```js
const Grid = styled.div`
  ${grid({halign: 'justify-center'})};
`;

const Quarter = styled.div`
  ${unit({size: 1/4})};
`;

const Eighth = styled.div`
  ${unit({size: 1/8})};
`;

const QuarterContent = styled.div`
  text-align: center;
  background: tomato;
  ${body({size: 'xl'})};
`;

const EighthContent = styled.div`
  text-align: center;
  background: peachpuff;
`;

render(
  <Grid>
    <Eighth><EighthContent> ⅛ </EighthContent></Eighth>
    <Quarter><QuarterContent> ¼ </QuarterContent></Quarter>
    <Eighth><EighthContent> ⅛ </EighthContent></Eighth>
  </Grid>
)
```

## Component

```
import Grid from '@lendi-ui/grid';
```

```js
const HalfContent = styled.div`
  text-align: center;
  background: tomato;
  ${body({size: 'xl'})};
`;

const QuarterContent = styled.div`
  text-align: center;
  background: peachpuff;
`;

render(
  <>
    <Grid halign={{sm: "right", xl: "justify-center"}}>
      <Grid.Unit size={1/4}><QuarterContent> ¼ </QuarterContent></Grid.Unit>
      <Grid.Unit size={1/2}><HalfContent> ½ </HalfContent></Grid.Unit>
      <Grid.Unit size={1/4}><QuarterContent> ¼ </QuarterContent></Grid.Unit>
    </Grid>

    <Grid halign={{sm: "right", xl: "justify-center"}} gutter={{xs: 'nil', md: 'md', xl: 'xl'}}>
      <Grid.Unit size={1/4}><QuarterContent> ¼ </QuarterContent></Grid.Unit>
      <Grid.Unit size={1/2}><HalfContent> ½ </HalfContent></Grid.Unit>
      <Grid.Unit size={1/4}><QuarterContent> ¼ </QuarterContent></Grid.Unit>
    </Grid>
  </>
);
```
