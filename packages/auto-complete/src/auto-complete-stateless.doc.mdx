## Usage
```
import { AutoCompleteStateless  } from '@lendi-ui/auto-complete';
```

```js
class PlaygoundComponent extends React.Component{
  constructor() {
    this.state = {
      value: '',
      dataSource: [],
    };
  }

  render() {
    return (
      <>
      <Heading size="xs">Search keyword like - 'Bank' - for more please see the working examples. <u>There are some positioning issues with dropdown on this playground page.</u></Heading>
        <AutoCompleteStateless
          dataSource={this.state.dataSource}
          value={this.state.value}
          onChange={(e) => {
            this.setState({ value: e.target.value, dataSource: this.getStaticData(e.target.value) });
          }}
          onSelect={(value) => this.setState({ value })}
          isFullWidth={false}
          onReset={() => {
            this.setState({
              value: '',
              dataSource: [],
            });
          }}
        />
      </>
    );
  }

  getStaticData(value) {
    return [{label: 'AMP Bank',value: 'AMP',},{label: 'ANZ',value: 'anz',},{label: 'Adelaide Bank',value: 'ADE',}]
          .filter(
              (data) => data.label.toLowerCase().indexOf(value.toLowerCase()) > -1
            );
  }
  
}

render(<PlaygoundComponent/>)


```
