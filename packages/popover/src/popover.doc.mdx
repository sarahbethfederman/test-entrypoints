import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import Popover from '@lendi-ui/popover';
```

```jsx
<>
  <Popover position='top'>
    <Popover.Target>
      <Info color="primary.500" />
    </Popover.Target>

    <Popover.Content>
      Example Text!
    </Popover.Content>
  </Popover>

  <br />
  <br />

  <Popover position='bottom'>
    <Popover.Target>
      Text can be a target as well
    </Popover.Target>

    <Popover.Content>
      Example Text!
    </Popover.Content>
  </Popover>
</>
```
### Design Notes
Popovers are meant to give related additional content. They can be much more verbose (compared to Tooltips) and they can include a header and many lines of text in the body.

## Properties

<PropTable>
  <PropTable.Entry required name="Popover.Content" type="string | React.ReactNode" description="Popover content, if it is a string it will come with default padding of small, otherwise consumer must style their content entirely."/>
  <PropTable.Entry name="position" type="{'top' | 'top-end' | 'top-start' | 'bottom' | 'bottom-end' | 'bottom-start' | 'left' | 'left-end' | 'left-start' | 'right' | 'right-end' | 'right-start'}" defaultValue="top" description="Popover content display position comparing with target component"/>
  <PropTable.Entry required name="Popover.Target" type="string | React.ReactNode" description="Target component that trigger Popover by clicking it"/>
