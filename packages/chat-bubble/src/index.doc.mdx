## Usage

```
import ChatBubble from '@lendi-ui/chat-bubble';
```

```js
const ExampleChatBubble = styled(ChatBubble)`
    ${mb('xs')}
    ${ml('md')}
    ${mr('lg')}
`;

const CustomerContent = styled(ChatBubble.Content)`
    ${bg('primary.500')} ${fg('shade.0')}
`;

const ContentBodyWrapper = styled(Body)`
    ${p('xs')}
`;

const BrokerContent = styled(ChatBubble.Content)`
    ${bg('shade.100')} ${fg('shade.700')}
`;

const PlaygroundCarousel = () => (
  <>
    <ExampleChatBubble direction='left'>
      <BrokerContent><ContentBodyWrapper size='sm'>Hey this is Lendi. Nice to meet you!</ContentBodyWrapper></BrokerContent>
    </ExampleChatBubble>
    <ExampleChatBubble direction='right'>
      <CustomerContent><ContentBodyWrapper size='sm'>Hi there, Iâ€™d like to know something, can you help me to get a loan?</ContentBodyWrapper></CustomerContent>
      <ChatBubble.Footer><Body size='xs' color='shade.200' mt='xxxs'>12:35PM Read</Body></ChatBubble.Footer>
    </ExampleChatBubble>
    <ExampleChatBubble direction='right' hasTail={false}>
      <CustomerContent><ContentBodyWrapper size='sm'>Without the tail</ContentBodyWrapper></CustomerContent>
    </ExampleChatBubble>
  </>
);

render(<PlaygroundCarousel />);

```

## Properties

<PropTable>
  <PropTable.Entry name="direction" type="{type Direction = 'left' | 'right';}" defaultValue="left" description="The side direction that ChatBubble component will rendered in the window."/>
  <PropTable.Entry name="hasTail" type="boolean" defaultValue="true" description="Prop to conditionally show the tail(top edge) on the chat bubble."/>
</PropTable>
