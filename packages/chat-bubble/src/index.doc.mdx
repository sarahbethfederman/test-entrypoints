## Usage

```
import ChatBubble from '@lendi-ui/chat-bubble';
```

```js
const CustomerChatBubble = styled(ChatBubble)`
    ${mb('xs')}
    ${ml('md')}
    ${mr('lg')}
`;

const CustomerContent = styled(ChatBubble.Content)`
    ${bg('primary.500')} ${fg('shade.0')}
`;

const ContentBodyWrapper = styled(Body)`
    ${p('xs')}
`;

const PlaygroundCarousel = () => (
  <CustomerChatBubble direction='right'>
    <CustomerContent><ContentBodyWrapper size='sm'>Hi there, Iâ€™d like to know something, can you help me to get a loan?</ContentBodyWrapper></CustomerContent>

```
