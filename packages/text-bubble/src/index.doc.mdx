import { PropTable } from '@lendi-ui/doc-utils';

## Usage

```
import TextBubble from '@lendi-ui/TextBubble';
```

```jsx
<>
<TextBubble>
  Your loan has been approved!
</TextBubble>

<TextBubble>
  Your loan has been approved! This is also an example of how it looks when there are multiple lines!
</TextBubble>

<TextBubble label="The content of this label is longer than the content text, so it will expand the bubble and wrap when appropriate">
  Your loan has been approved!
</TextBubble>

<TextBubble hasOverline={false}>
  Rendered without any overline text.
</TextBubble>

<TextBubble isFullWidth>
  Using full width.
</TextBubble>
</>
```

## Properties

<PropTable>
  <PropTable.Entry
    name="label"
    type="string"
    defaultValue={"Timestamp"}
    description="Label to appear above the text bubble, or else defaults to a current timestamp"
  />
  <PropTable.Entry
    name="children"
    type="React.ReactNode"
    description="Content of the Text Bubble"
  />
  <PropTable.Entry
    name="hasOverline"
    type="boolean"
    defaultValue={true}
    description="Flag for whether or not to include the overline text above. The label property will be ignored if this is disabled"
  />
  <PropTable.Entry
    name="isFullWidth"
    type="boolean"
    defaultValue={false}
    description="Flag for whether or not the element should use the full width of the parent container"
  />
</PropTable>
