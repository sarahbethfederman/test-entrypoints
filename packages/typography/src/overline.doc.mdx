# overline

Styles for overline copy.

## Mixin

```
import {overline} from '@lendi-ui/overline';
```

```js
const Text = styled.div`
  ${overline({size: 'lg', colorScheme: 'light'})}
`;

render(<Text>Default overline</Text>)
```

### Options

<PropTable>
  <PropTable.Entry name="size" type={'"lg | "md" | "sm"'} required description="The overline copy size."/>
  <PropTable.Entry name="colorScheme" type={'"light" | "dark"'} default="light" description="Determines the colour scheme for the overline copy. If no value is specified, defaults to 'light'. Light should be implemented on light backgrounds. Dark should be implemented on dark backgrounds."/>
  <PropTable.Entry name="align" type={'"left" | "center" | right"'} default="left" description="Horizontal alignment of overline copy. If value is not provided, this defaults to left text alignment."/>
</PropTable>

import {Overline} from '@lendi-ui/typography';
