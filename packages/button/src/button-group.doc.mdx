import {PropTable} from '@lendi-ui/doc-utils';
import { Lock } from '@lendi-ui/icon'

## Usage

```
import {ButtonGroup} from '@lendi-ui/button';
```

```jsx
<>
<ButtonGroup size="xs">
  <ButtonGroup.Button variant="secondary">&lt;</ButtonGroup.Button>
  <ButtonGroup.Button variant="primary">Continue</ButtonGroup.Button>
</ButtonGroup>
<br /><br />
<ButtonGroup size="sm">
  <ButtonGroup.Button variant="secondary">&lt;</ButtonGroup.Button>
  <ButtonGroup.Button variant="primary">Continue</ButtonGroup.Button>
</ButtonGroup>
<br /><br />
<ButtonGroup size={{mobile:'sm', tablet:'md', desktop:'lg'}}>
  <ButtonGroup.Button variant="secondary">&lt;</ButtonGroup.Button>
  <ButtonGroup.Button variant="primary">Continue</ButtonGroup.Button>
</ButtonGroup>
<br /><br />
<ButtonGroup size={'sm'}>
  <ButtonGroup.IconButton icon={Lock} />
  <ButtonGroup.IconButton icon={Lock} />
</ButtonGroup>
</>
```
