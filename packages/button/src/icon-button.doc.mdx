import { PropTable } from '@lendi-ui/doc-utils';
import { Lock } from '@lendi-ui/icon';


## Usage

```
import { IconButton } from '@lendi-ui/button';
```

```jsx
<>
    <IconButton onClick={(event:React.SyntheticEvent) => alert('Have a nice day!')} size='md' aria-label="Lock" icon={Lock} />
</>
```

## Properties

<PropTable>
  <PropTable.Entry name="color" type="string" defaultValue="primary.500" description="The colour of the icon"/>
  <PropTable.Entry name="href" type="string" description="URL to link to"/>
  <PropTable.Entry name="icon" type="React.ComponentClass" description="The icon to be displayed eg. {Lock}. Please note that the icon used needs to be imported"/>
  <PropTable.Entry name="isDisabled" type="boolean" defaultValue={false} description="Whether the button is disabled"/>
  <PropTable.Entry name="isLoading" type="boolean" defaultValue={false} description="Whether the button is loading. Swaps the icon out for a spinner"/>
  <PropTable.Entry name="onClick" type="(event:React.SyntheticEvent) => void" description="Handle the click event"/>
  <PropTable.Entry name="size" type={`"xs" |"sm" | "md" | "lg"`} defaultValue="md" description="The button size"/>
</PropTable>


### Notes
<b>Note: On top of the properties listed in the table above, this component also supports all native Aria attributes, HTML global data-* attributes, and standard HTML attributes (eg. title, className, id, role, itemProp, itemID, itemRef etc.)</b>
