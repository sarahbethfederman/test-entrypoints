import { PropTable } from '@lendi-ui/doc-utils';

## Usage

```
import Spinner from '@lendi-ui/spinner';
```

```jsx
<>
  <Spinner variant="dark" />
  <br />
  <br />
  <Button variant="primary" size="md" after={<Spinner variant="light" />}>
    Button
  </Button>
  <br />
  <br />
  <Button variant="secondary" size="md" before={<Spinner variant="dark" />}>
    Button
  </Button>
</>
```
