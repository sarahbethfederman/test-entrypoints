## Usage

```
import { Select } from '@lendi-ui/dropdown';

```

```js

const items = [{
                label: 'AMP Bank',
                value: 'AMP',
              },
              {
                label: 'ANZ',
                value: 'anz',
              },
              {
                label: 'Adelaide Bank',
                value: 'ADE',
              },
              {
                label: 'Adelaide Bank (Commercial) - a very very very long name',
                value: 'ADEC',
              }]


class PlaygoundComponent extends React.Component{

  constructor() {
    this.state = {
      values: [
        {
          label: 'Adelaide Bank',
          value: 'ADE',
        },
      ],
      inputValue: '',
    }
  }
  
  render() {
    return (
        <Select
          size="sm"
          isClearable={true}
          options={items}
          isFullWidth={true}
          placeholder="Multi-Select"
          onChangeItem={(values) => this.setState({values})}
          value={this.state.values}
          isClearableByBackspace={true}
          onFocus={() => console.log('onFocus')}
          onBlur={() => console.log('onBlur')}
          hideSelectedOptions={false}
          isError={false}
          isMultiple
          inputValue={this.state.inputValue}
          onInputChange={(inputValue) => {
            this.setState({
              inputValue,
            });
          }}
        />
    );
  }

  
}

render(<PlaygoundComponent/>)


```
