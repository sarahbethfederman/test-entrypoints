import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import Dropdown from '@lendi-ui/dropdown';
```

```jsx

items= [
    {
      value: '1',
      label: '5 years'
    },
    {
      value: '2',
      label: '10 years'
    },
    {
      value: '3',
      label: '15 years'
    },
    {
      value: '4',
      label: '20 years'
    },
    {
      value: '5',
      label: '25 years'
    }
],

 <Dropdown
    size={{mobile:'sm', tablet:'md', desktop:'lg'}}
    items={items}
    isFullWidth={true}
    value='2'
/>

```
