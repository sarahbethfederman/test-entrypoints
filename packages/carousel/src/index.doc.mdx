import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import Carousel from '@lendi-ui/carousel';
```

```jsx
<Carousel>
  <div> #1 </div>
  <div> #2 </div>
  <div> #3 </div>
</Carousel>
```
