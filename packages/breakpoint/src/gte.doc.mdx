
# gte(breakpoint)

Apply styles at a size larger or equal to the specified breakpoint.

## Import
```
import { gte, useBreakpoint, getBreakpoint } from '@lendi-ui/breakpoint';
```

## Usage

```js

const common = css`
  display: block;
  ${p('sm')};
  ${body};
  ${fg('shade.800')};
  ${bg('shade.300')};
`;

const ExtraSmall = styled.code`
  ${common};
  ${gte('xs')`
    ${fg('shade.0')};
    ${bg('primary.900')};
  `};
`;

const Small = styled.code`
  ${common};
  ${gte('sm')`
    ${fg('shade.0')};
    ${bg('primary.700')};
  `};
`;

const Medium = styled.code`
  ${common};
  ${gte('md')`
    ${bg('primary.500')};
  `}
`;

const Large = styled.code`
  ${common};
  ${gte('lg')`
    ${bg('primary.300')};
  `}
`

const ExtraLarge = styled.code`
  ${common};
  ${gte('xl')`
    ${bg('primary.100')};
  `};
`;

const CurrentBreakpoint = () => {
  const [breakpoint, setBreakpoint] = React.useState('xs');

  useBreakpoint(setBreakpoint, undefined, true);

  return (<>{breakpoint}</>);
}

render(
  <>
    ğŸ‘‡ Resize the window to see the color change at each breakpoint:
    <ExtraSmall>gte('xs')</ExtraSmall>
    <Small>gte('sm')</Small>
    <Medium>gte('md')</Medium>
    <Large>gte('lg')</Large>
    <ExtraLarge>gte('xl')</ExtraLarge>
    <Body>Breakpoint on page load is: {getTshirtBreakpoint()}</Body>
    <Body>The current breakpoint is: <CurrentBreakpoint /></Body>
  </>
);
```
