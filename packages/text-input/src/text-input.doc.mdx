import {PropTable} from '@lendi-ui/doc-utils';

## Usage

```
import Input from '@lendi-ui/text-input';
```

```jsx

<>
 <Input
    size={'md'}
    placeholder="Input here ..."
    onChange={() =>{}}
    isFullWidth={true}
    before={<Email color='primary.500' height="26px" style={{marginLeft: "16px"}} />}
 />
 </>
```
