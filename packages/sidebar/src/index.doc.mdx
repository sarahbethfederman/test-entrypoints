import { PropTable } from '@lendi-ui/doc-utils';

## Usage

```
import Sidebar from '@lendi-ui/Sidebar';
```

```jsx
// Note: setting the prop "isVisible" to "true" will not open the Sidebar here. For a working example you can go to the example page, linked above.
<Sidebar
    isVisible={false}
 />
             
```
