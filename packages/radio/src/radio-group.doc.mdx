# Usage

```
import {RadioGroup} from '@lendi-ui/radio';
```

```js
class PlaygroundComponent extends React.Component{
  constructor() {
    this.state = {
      selected: 1
    }
    this.onChange = this.onChange.bind(this);
  }

  onChange(e) {
    this.setState({selected: e.target.value});
  }

  render() {
    return (
      <RadioGroup onChange={this.onChange} selectedValue={this.state.selected}>
        <RadioGroup.Radio
          label='LUI Radio Group option 1'
          value='1'
        />
        <RadioGroup.Radio
          label='LUI Radio Group option 2'
          value='2'
        />
        <RadioGroup.Radio
          label='LUI Radio Group option 3'
          value='3'
        />
      </RadioGroup>
    );
  }
}

render(<PlaygroundComponent/>)
```
